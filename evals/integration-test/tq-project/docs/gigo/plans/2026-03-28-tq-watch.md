# tq watch — Implementation Plan

> **For agentic workers:** Use gigo:execute to implement this plan task-by-task.
> Steps use checkbox (`- [ ]`) syntax for tracking.

**Spec:** `docs/gigo/specs/2026-03-28-tq-watch-design.md`

**Goal:** Add BoltDB persistence and a `tq watch` command that streams task state changes in real-time.

**Architecture:** Two sub-projects — BoltDB persistence (expand Store interface, implement BoltStore, wire factory) then watch command (poll-diff-emit loop with filter, NDJSON, and signal handling). Watch opens DB read-only to coexist with writers.

**Tech Stack:** Go 1.22, Cobra, go.etcd.io/bbolt

```go
package store

import (
    "crypto/rand"
    "encoding/hex"
    "encoding/json"
    "fmt"
    "time"

    bbolt "go.etcd.io/bbolt"
)

var tasksBucket = []byte("tasks")

// BoltStore is a BoltDB-backed Store implementation.
type BoltStore struct {
    db *bbolt.DB
}

// NewBoltStore opens a read-write BoltDB at path.
func NewBoltStore(path string) (*BoltStore, error) {
    db, err := bbolt.Open(path, 0o600, &bbolt.Options{Timeout: 1 * time.Second})
    if err != nil {
        return nil, fmt.Errorf("open bolt db: %w", err)
    }
    err = db.Update(func(tx *bbolt.Tx) error {
        _, err := tx.CreateBucketIfNotExists(tasksBucket)
        return err
    })
    if err != nil {
        db.Close()
        return nil, fmt.Errorf("create tasks bucket: %w", err)
    }
    return &BoltStore{db: db}, nil
}

// NewBoltStoreReadOnly opens a read-only BoltDB at path.
// Does not acquire the write lock — coexists with writers.
func NewBoltStoreReadOnly(path string) (*BoltStore, error) {
    db, err := bbolt.Open(path, 0o600, &bbolt.Options{
        Timeout:  1 * time.Second,
        ReadOnly: true,
    })
    if err != nil {
        return nil, fmt.Errorf("open bolt db read-only: %w", err)
    }
    return &BoltStore{db: db}, nil
}
```

- [ ] **Step 3: Implement `Add` and `Get` on BoltStore**

```go
func (b *BoltStore) Add(task Task) error {
    if task.ID == "" {
        buf := make([]byte, 4)
        if _, err := rand.Read(buf); err != nil {
            return err
        }
        task.ID = hex.EncodeToString(buf)
    }
    return b.db.Update(func(tx *bbolt.Tx) error {
        data, err := json.Marshal(task)
        if err != nil {
            return err
        }
        return tx.Bucket(tasksBucket).Put([]byte(task.ID), data)
    })
}

func (b *BoltStore) Get(id string) (Task, error) {
    var task Task
    err := b.db.View(func(tx *bbolt.Tx) error {
        bkt := tx.Bucket(tasksBucket)
        if bkt == nil {
            return ErrNotFound // no bucket = no tasks (read-only DB without prior writes)
        }
        data := bkt.Get([]byte(id))
        if data == nil {
            return ErrNotFound
        }
        return json.Unmarshal(data, &task)
    })
    return task, err
}
```

- [ ] **Step 4: Implement `List` on BoltStore**

```go
func (b *BoltStore) List(filter Filter) ([]Task, error) {
    var result []Task
    err := b.db.View(func(tx *bbolt.Tx) error {
        bkt := tx.Bucket(tasksBucket)
        if bkt == nil {
            return nil // no bucket = no tasks (read-only DB without prior writes)
        }
        return bkt.ForEach(func(k, v []byte) error {
            var t Task
            if err := json.Unmarshal(v, &t); err != nil {
                return err
            }
            if filter.State != nil && t.State != *filter.State {
                return nil
            }
            result = append(result, t)
            return nil
        })
    })
    return result, err
}
```

- [ ] **Step 5: Implement `Transition` on BoltStore**

Compare-and-swap within a single read-write transaction for atomicity:

```go
func (b *BoltStore) Transition(id string, from, to State) error {
    return b.db.Update(func(tx *bbolt.Tx) error {
        bkt := tx.Bucket(tasksBucket)
        data := bkt.Get([]byte(id))
        if data == nil {
            return ErrNotFound
        }
        var task Task
        if err := json.Unmarshal(data, &task); err != nil {
            return err
        }
        if task.State != from {
            return ErrConflict
        }
        task.State = to
        updated, err := json.Marshal(task)
        if err != nil {
            return err
        }
        return bkt.Put([]byte(id), updated)
    })
}
```

```go
        if bkt.Get([]byte(id)) == nil {
            return ErrNotFound
```

```go
func resolveQueueDir() (string, error) {
    dir := tqDir
    if dir == "" {
        dir = os.Getenv("TQ_DIR")
    }
    if dir == "" {
        home, err := os.UserHomeDir()
        if err != nil {
            return "", fmt.Errorf("tq: cannot determine home directory: %w", err)
        }
        dir = filepath.Join(home, ".tq")
    }
    return dir, nil
}

func openStore() (store.Store, error) {
    dir, err := resolveQueueDir()
    if err != nil {
        return nil, err
    }
    if err := os.MkdirAll(dir, 0o755); err != nil {
        return nil, fmt.Errorf("tq: cannot create queue directory %q: %w", dir, err)
    }
    return store.NewBoltStore(filepath.Join(dir, "queue.db"))
}

func openStoreReadOnly() (store.Store, error) {
    dir, err := resolveQueueDir()
    if err != nil {
        return nil, err

    dbPath := filepath.Join(dir, "queue.db")
    // If the DB file doesn't exist yet (no tq add has ever run),
    // return an empty MemoryStore — nothing to watch.
    if _, err := os.Stat(dbPath); os.IsNotExist(err) {
        return store.NewMemoryStore(), nil
```

- [ ] **Step 2: Implement filter parsing**

```go
func parseWatchFilter(raw string) (watchFilterSpec, error) {
    spec := watchFilterSpec{
        IDs:   make(map[string]bool),
        Froms: make(map[string]bool),
        Tos:   make(map[string]bool),
    }
    if raw == "" {
        return spec, nil
    }
    for _, part := range strings.Split(raw, ",") {
        kv := strings.SplitN(part, "=", 2)
        if len(kv) != 2 || kv[0] == "" || kv[1] == "" {
            return spec, fmt.Errorf("tq: invalid filter expression %q (expected key=value)", part)
        }
        key, val := kv[0], kv[1]
        switch key {
        case "id":
            spec.IDs[val] = true
        case "from":
            if !validState(val) {
                return spec, fmt.Errorf("tq: invalid state %q in filter (valid: %s)", val, stateList())
            }
            spec.Froms[val] = true
        case "to":
            if val != "deleted" && !validState(val) {
                return spec, fmt.Errorf("tq: invalid state %q in filter (valid: %s, deleted)", val, stateList())
            }
            spec.Tos[val] = true
        default:
            return spec, fmt.Errorf("tq: unknown filter key %q (valid: id, from, to)", key)
        }
    }
    return spec, nil
}

func validState(s string) bool {
    for _, st := range store.AllStates {
        if string(st) == s {
            return true
        }
    }
    return false
}

func stateList() string {
    names := make([]string, len(store.AllStates))
    for i, s := range store.AllStates {
        names[i] = string(s)
    }
    return strings.Join(names, ", ")
}
```

```go
func matchesFilter(ev WatchEvent, spec watchFilterSpec) bool {
    if len(spec.IDs) > 0 && !spec.IDs[ev.TaskID] {
        return false
    }
    if len(spec.Froms) > 0 && !spec.Froms[ev.From] {
        return false
    }
    if len(spec.Tos) > 0 && !spec.Tos[ev.To] {
        return false
    }
    return true
}

func printEvent(ev WatchEvent, jsonOut bool, w io.Writer) {
    if jsonOut {
        json.NewEncoder(w).Encode(ev)
        return
    }
    from := ev.From
    if from == "" {
        from = "NEW"
    }
    fmt.Fprintf(w, "%s  %s  %s->%s  %s\n", ev.Timestamp, ev.TaskID, from, ev.To, ev.Name)
}
```

```go
func runWatchLoop(ctx context.Context, open storeOpener, spec watchFilterSpec, interval time.Duration, jsonOut bool, w io.Writer, snapshotReady chan struct{}) error {
    // Take initial snapshot
    snapshot := make(map[string]snapshotEntry)
    s, err := open()
    if err != nil {
        return fmt.Errorf("tq: cannot open queue: %w", err)
    }
    tasks, err := s.List(store.Filter{})
    s.Close()
    if err != nil {
        return fmt.Errorf("tq: cannot read queue: %w", err)
    }
    for _, t := range tasks {
        snapshot[t.ID] = snapshotEntry{State: t.State, Name: t.Name}
    }

    // Signal that snapshot is ready (for tests)
    if snapshotReady != nil {
        close(snapshotReady)
    }

    ticker := time.NewTicker(interval)
    defer ticker.Stop()

    for {
        select {
        case <-ctx.Done():
            return nil
        case <-ticker.C:
            s, err := open()
            if err != nil {
                return fmt.Errorf("tq: cannot open queue: %w", err)
            }
            tasks, err := s.List(store.Filter{})
            s.Close()
            if err != nil {
                return fmt.Errorf("tq: cannot read queue: %w", err)
            }

            // Build set of current IDs
            currentIDs := make(map[string]bool, len(tasks))
            for _, t := range tasks {
                currentIDs[t.ID] = true
                entry, existed := snapshot[t.ID]
                if !existed {
                    // New task
                    ev := WatchEvent{
                        Timestamp: time.Now().UTC().Format(time.RFC3339),
                        TaskID:    t.ID,
                        Name:      t.Name,
                        From:      "",
                        To:        string(t.State),
                    }
                    if matchesFilter(ev, spec) {
                        printEvent(ev, jsonOut, w)
                    }
                } else if t.State != entry.State {
                    // State transition
                    ev := WatchEvent{
                        Timestamp: time.Now().UTC().Format(time.RFC3339),
                        TaskID:    t.ID,
                        Name:      t.Name,
                        From:      string(entry.State),
                        To:        string(t.State),
                    }
                    if matchesFilter(ev, spec) {
                        printEvent(ev, jsonOut, w)
                    }
                }
                snapshot[t.ID] = snapshotEntry{State: t.State, Name: t.Name}
            }

            // Detect deletions
            for id, entry := range snapshot {
                if !currentIDs[id] {
                    ev := WatchEvent{
                        Timestamp: time.Now().UTC().Format(time.RFC3339),
                        TaskID:    id,
                        Name:      entry.Name,
                        From:      string(entry.State),
                        To:        "deleted",
                    }
                    if matchesFilter(ev, spec) {
                        printEvent(ev, jsonOut, w)
                    }
                    delete(snapshot, id)
                }
            }
        }
    }
}
```
